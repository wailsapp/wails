# Автоматические обновления

Wails v3 предоставляет встроенную систему обновлений, которая поддерживает автоматическую проверку, загрузку и установку обновлений. Система обновлений включает поддержку дельта-обновлений бинарных файлов (патчей) для минимизации размера загружаемых данных.

## Быстрый старт

Добавьте сервис обновлений в ваше приложение:

```go title="main.go"
package main

import (
    "github.com/wailsapp/wails/v3/pkg/application"
    "time"
)

func main() {
    // Создаём сервис обновлений
    updater, err := application.CreateUpdaterService(
        "1.0.0", // Текущая версия
        application.WithUpdateURL("https://updates.example.com/myapp/"),
        application.WithCheckInterval(24 * time.Hour),
    )
    if err != nil {
        panic(err)
    }

    app := application.New(application.Options{
        Name: "MyApp",
        Services: []application.Service{
            application.NewService(updater),
        },
    })

    // ... остальная часть вашего приложения
    app.Run()
}
```

Затем используйте его из фронтенда:

```typescript title="App.tsx"
import { updater } from './bindings/myapp';

async function checkForUpdates() {
  const update = await updater.CheckForUpdate();

  if (update) {
    console.log(`Доступна новая версия: ${update.version}`);
    console.log(`Примечания к выпуску: ${update.releaseNotes}`);

    // Загрузка и установка
    await updater.DownloadAndApply();
  }
}
```

## Параметры конфигурации

Система обновлений поддерживает различные параметры конфигурации:

```go
updater, err := application.CreateUpdaterService(
    "1.0.0",
    // Обязательно: URL, где размещены манифесты обновлений
    application.WithUpdateURL("https://updates.example.com/myapp/"),

    // Необязательно: Автоматически проверять обновления каждые 24 часа
    application.WithCheckInterval(24 * time.Hour),

    // Необязательно: Разрешить предварительные версии (pre-release)
    application.WithAllowPrerelease(true),

    // Необязательно: Канал обновлений (stable, beta, canary)
    application.WithChannel("stable"),

    // Необязательно: Требовать подписанные обновления
    application.WithRequireSignature(true),
    application.WithPublicKey("your-ed25519-public-key"),
)
```

## API фронтенда

Система обновлений предоставляет методы, которые автоматически связываются с вашим фронтендом:

### Типы TypeScript

```typescript
interface UpdateInfo {
  version: string;
  releaseDate: Date;
  releaseNotes: string;
  size: number;
  patchSize?: number;
  mandatory: boolean;
  hasPatch: boolean;
}

interface Updater {
  // Получить текущую версию приложения
  GetCurrentVersion(): string;

  // Проверить наличие доступного обновления
  CheckForUpdate(): Promise<UpdateInfo | null>;

  // Загрузить обновление (отправляет события прогресса)
  DownloadUpdate(): Promise<void>;

  // Применить загруженное обновление (перезапускает приложение)
  ApplyUpdate(): Promise<void>;

  // Загрузить и применить за один вызов
  DownloadAndApply(): Promise<void>;

  // Получить текущее состояние: idle, checking, available, downloading, ready, installing, error
  GetState(): string;

  // Получить информацию о доступном обновлении
  GetUpdateInfo(): UpdateInfo | null;

  // Получить последнее сообщение об ошибке
  GetLastError(): string;

  // Сбросить состояние системы обновлений
  Reset(): void;
}
```

### События прогресса

Подписывайтесь на события прогресса загрузки:

```typescript
import { Events } from '@wailsio/runtime';

Events.On('updater:progress', (data) => {
  console.log(`Загружено: ${data.downloaded} / ${data.total}`);
  console.log(`Прогресс: ${data.percentage.toFixed(1)}%`);
  console.log(`Скорость: ${(data.bytesPerSecond / 1024 / 1024).toFixed(2)} МБ/с`);
});
```

## Полный пример

Вот полный пример с компонентом React:

```tsx title="UpdateChecker.tsx"
import { useState, useEffect } from 'react';
import { updater } from './bindings/myapp';
import { Events } from '@wailsio/runtime';

interface Progress {
  downloaded: number;
  total: number;
  percentage: number;
  bytesPerSecond: number;
}

export function UpdateChecker() {
  const [checking, setChecking] = useState(false);
  const [updateInfo, setUpdateInfo] = useState<any>(null);
  const [downloading, setDownloading] = useState(false);
  const [progress, setProgress] = useState<Progress | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    // Подписываемся на события прогресса
    const cleanup = Events.On('updater:progress', (data: Progress) => {
      setProgress(data);
    });

    // Проверяем наличие обновлений при монтировании
    checkForUpdates();

    return () => cleanup();
  }, []);

  async function checkForUpdates() {
    setChecking(true);
    setError(null);

    try {
      const info = await updater.CheckForUpdate();
      setUpdateInfo(info);
    } catch (err) {
      setError(err.message);
    } finally {
      setChecking(false);
    }
  }

  async function downloadAndInstall() {
    setDownloading(true);
    setError(null);

    try {
      await updater.DownloadAndApply();
      // Приложение перезапустится автоматически
    } catch (err) {
      setError(err.message);
      setDownloading(false);
    }
  }

  if (checking) {
    return <div>Проверка обновлений...</div>;
  }

  if (error) {
    return (
      <div>
        <p>Ошибка: {error}</p>
        <button onClick={checkForUpdates}>Повторить</button>
      </div>
    );
  }

  if (!updateInfo) {
    return (
      <div>
        <p>У вас установлена последняя версия! (v{updater.GetCurrentVersion()})</p>
        <button onClick={checkForUpdates}>Проверить снова</button>
      </div>
    );
  }

  if (downloading) {
    return (
      <div>
        <p>Загрузка обновления...</p>
        {progress && (
          <div>
            <progress value={progress.percentage} max={100} />
            <p>{progress.percentage.toFixed(1)}%</p>
            <p>{(progress.bytesPerSecond / 1024 / 1024).toFixed(2)} МБ/с</p>
          </div>
        )}
      </div>
    );
  }return (
    <div>
      <h3>Доступно обновление!</h3>
      <p>Доступна версия {updateInfo.version}</p>
      <p>Размер: {updateInfo.hasPatch
        ? `${(updateInfo.patchSize / 1024).toFixed(0)} КБ (патч)`
        : `${(updateInfo.size / 1024 / 1024).toFixed(1)} МБ`}
      </p>
      <div dangerouslySetInnerHTML={{ __html: updateInfo.releaseNotes }} />
      <button onClick={downloadAndInstall}>Скачать и установить</button>
      <button onClick={() => setUpdateInfo(null)}>Пропустить</button>
    </div>
  );
}
```

## Стратегии обновлений

### Проверка при запуске

```go
func (a *App) OnStartup(ctx context.Context) {
    // Проверка обновлений через короткую задержку
    go func() {
        time.Sleep(5 * time.Second)
        info, err := a.updater.CheckForUpdate()
        if err == nil && info != nil {
            // Отправка события во фронтенд
            application.Get().EmitEvent("update-available", info)
        }
    }()
}
```

### Фоновая проверка

Настройте автоматическую фоновую проверку:

```go
updater, _ := application.CreateUpdaterService(
    "1.0.0",
    application.WithUpdateURL("https://updates.example.com/myapp/"),
    application.WithCheckInterval(6 * time.Hour), // Проверка каждые 6 часов
)
```

### Пункт меню ручной проверки

```go
menu := application.NewMenu()
menu.Add("Проверить обновления...").OnClick(func(ctx *application.Context) {
    info, err := updater.CheckForUpdate()
    if err != nil {
        application.InfoDialog().SetMessage("Ошибка проверки обновлений").Show()
        return
    }
    if info == nil {
        application.InfoDialog().SetMessage("У вас установлена последняя версия!").Show()
        return
    }
    // Показать диалог обновления...
})
```

## Дельта-обновления (патчи)

Дельта-обновления (патчи) позволяют пользователям скачивать только изменения между версиями, что значительно уменьшает размер загрузки.

### Как это работает

1. При сборке новой версии создайте патчи на основе предыдущих версий
2. Разместите патчи вместе с полными обновлениями на вашем сервере
3. Система обновлений автоматически скачивает патчи, если они доступны
4. Если применение патча не удалось, происходит возврат к полной загрузке

### Генерация патчей

Патчи генерируются с использованием алгоритма bsdiff. Вам понадобится инструмент `bsdiff`:

```bash
# Установка bsdiff (macOS)
brew install bsdiff

# Установка bsdiff (Ubuntu/Debian)
sudo apt-get install bsdiff

# Генерация патча
bsdiff old-binary new-binary patch.bsdiff
```

### Именование файлов патчей

Организуйте ваши патчи в каталоге обновлений:

```
updates/
├── update.json
├── myapp-1.2.0-macos-arm64.tar.gz
├── myapp-1.2.0-windows-amd64.zip
└── patches/
    ├── 1.0.0-to-1.2.0-macos-arm64.patch
    ├── 1.1.0-to-1.2.0-macos-arm64.patch
    ├── 1.0.0-to-1.2.0-windows-amd64.patch
    └── 1.1.0-to-1.2.0-windows-amd64.patch
```

<Aside type="tip">
Храните патчи для последних нескольких версий. Пользователи очень старых версий автоматически скачают полное обновление.
</Aside>

## Размещение обновлений

### Размещение статических файлов

Обновления можно размещать на любом сервере статических файлов:

- **Amazon S3** / **Cloudflare R2**
- **Google Cloud Storage**
- **GitHub Releases**
- **Любой CDN или веб-сервер**

Пример структуры бакета S3:

```
s3://my-updates-bucket/myapp/
├── stable/
│   ├── update.json
│   ├── myapp-1.2.0-macos-arm64.tar.gz
│   └── patches/
│       └── 1.1.0-to-1.2.0-macos-arm64.patch
└── beta/
    └── update.json
```

### Настройка CORS

Если размещение происходит на другом домене, настройте CORS:

```json
{
  "CORSRules": [
    {
      "AllowedOrigins": ["*"],
      "AllowedMethods": ["GET"],
      "AllowedHeaders": ["*"]
    }
  ]
}
```

## Безопасность

### Проверка контрольных сумм

Все загрузки проверяются на соответствие контрольным суммам SHA256 в манифесте:

```json
{
  "checksum": "sha256:e3b0c44298fc1c149afbf4c8996fb92427ae41e4649b934ca495991b7852b855"
}
```

### Проверка подписей

Для дополнительной безопасности включите проверку подписей:

<Steps>
1. **Сгенерируйте пару ключей**:
   ```bash
   # Генерация пары ключей Ed25519
   openssl genpkey -algorithm Ed25519 -out private.pem
   openssl pkey -in private.pem -pubout -out public.pem
   ```

2. **Подпишите манифест обновления**:
   ```bash
   openssl pkeyutl -sign -inkey private.pem -in update.json -out update.json.sig
   ```

3. **Настройте систему обновлений**:
   ```go
   updater, _ := application.CreateUpdaterService(
       "1.0.0",
       application.WithUpdateURL("https://updates.example.com/myapp/"),
       application.WithRequireSignature(true),
       application.WithPublicKey("MCowBQYDK2VwAyEA..."), // Открытый ключ в формате Base64
   )
   ```
</Steps>

## Лучшие практики

### Следует

- Тщательно тестировать обновления перед развертыванием
- Сохранять предыдущие версии для возможности отката
- Показывать пользователям примечания к выпуску
- Позволять пользователям пропускать необязательные обновления
- Использовать HTTPS для всех загрузок
- Проверять контрольные суммы перед применением обновлений
- Корректно обрабатывать ошибки

### Не следует

- Принудительно перезапускать приложение без предупреждения
- Пропускать проверку контрольных сумм
- Прерывать пользователей во время важной работы
- Немедленно удалять предыдущую версию
- Игнорировать сбои обновлений

## Устранение неполадок

### Обновление не найдено

- Проверьте правильность URL манифеста
- Убедитесь, что ключ платформы совпадает (например, `macos-arm64`)
- Убедитесь, что версия в манифесте новее

### Сбой загрузки

- Проверьте сетевое подключение
- Убедитесь, что URL загрузки доступен
- Проверьте конфигурацию CORS, если запросы кросс-доменные

### Сбой патча

- Система обновлений автоматически переходит к полной загрузке
- Убедитесь, что `bspatch` доступен в системе
- Проверьте правильность контрольной суммы патча

### Приложение не перезапускается

- На macOS убедитесь, что приложение правильно подписано кодом
- На Windows проверьте наличие блокировок файлов
- На Linux проверьте права доступа к файлам
