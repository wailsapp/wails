在本教程中，你将构建一个笔记应用，展示如何使用 Wails v3 进行文件操作、原生对话框以及现代桌面应用模式的开发。

![笔记应用截图](../../../../assets/notes-app.png)

## 你将构建的内容

- 一个完整的笔记应用，具备创建、编辑和删除功能
- 用于导入和导出笔记的原生文件保存/打开对话框
- 基于防抖（debounce）的打字自动保存功能，以减少不必要的更新
- 模仿 Apple Notes 的专业双栏布局（侧边栏 + 编辑器）

## 你将学到的内容

- 在 Wails 中使用原生文件对话框（`SaveFileDialog`、`OpenFileDialog`、`InfoDialog`）
- 使用 JSON 文件进行数据持久化
- 实现防抖自动保存模式
- 使用现代 CSS 构建专业的桌面 UI
- 正确的 Go 结构体 JSON 序列化

2. **创建 NotesService**

   在项目根目录创建新文件 `notesservice.go`：

   ```go
   package main

   import (
   	"encoding/json"
   	"errors"
   	"os"
   	"time"

   	"github.com/wailsapp/wails/v3/pkg/application"
   )

   type Note struct {
   	ID        string    `json:"id"`
   	Title     string    `json:"title"`
   	Content   string    `json:"content"`
   	CreatedAt time.Time `json:"createdAt"`
   	UpdatedAt time.Time `json:"updatedAt"`
   }

   type NotesService struct {
   	notes []Note
   }

   func NewNotesService() *NotesService {
   	return &NotesService{
   		notes: make([]Note, 0),
   	}
   }

   // GetAll 返回所有笔记
   func (n *NotesService) GetAll() []Note {
   	return n.notes
   }

   // Create 创建新笔记
   func (n *NotesService) Create(title, content string) Note {
   	note := Note{
   		ID:        generateID(),
   		Title:     title,
   		Content:   content,
   		CreatedAt: time.Now(),
   		UpdatedAt: time.Now(),
   	}
   	n.notes = append(n.notes, note)
   	return note
   }

   // Update 更新现有笔记
   func (n *NotesService) Update(id, title, content string) error {
   	for i := range n.notes {
   		if n.notes[i].ID == id {
   			n.notes[i].Title = title
   			n.notes[i].Content = content
   			n.notes[i].UpdatedAt = time.Now()
   			return nil
   		}
   	}
   	return errors.New("note not found")
   }

   // Delete 删除笔记
   func (n *NotesService) Delete(id string) error {
   	for i := range n.notes {
   		if n.notes[i].ID == id {
   			n.notes = append(n.notes[:i], n.notes[i+1:]...)
   			return nil
   		}
   	}
   	return errors.New("note not found")
   }

   // SaveToFile 将笔记保存到文件
   func (n *NotesService) SaveToFile() error {
   	path, err := application.Get().Dialog.SaveFile().
   		SetFilename("notes.json").
   		AddFilter("JSON Files", "*.json").
   		PromptForSingleSelection()

   	if err != nil {
   		return err
   	}

   	data, err := json.MarshalIndent(n.notes, "", "  ")
   	if err != nil {
   		return err
   	}

   	if err := os.WriteFile(path, data, 0644); err != nil {
   		return err
   	}

   	application.Get().Dialog.Info().
   		SetTitle("Success").
   		SetMessage("Notes saved successfully!").
   		Show()

   	return nil
   }

   // LoadFromFile 从文件加载笔记
   func (n *NotesService) LoadFromFile() error {
   	path, err := application.Get().Dialog.OpenFile().
   		AddFilter("JSON Files", "*.json").
   		PromptForSingleSelection()

   	if err != nil {
   		return err
   	}

   	data, err := os.ReadFile(path)
   	if err != nil {
   		return err
   	}

   	var notes []Note
   	if err := json.Unmarshal(data, &notes); err != nil {
   		return err
   	}

   	n.notes = notes

   	application.Get().Dialog.Info().
   		SetTitle("Success").
   		SetMessage("Notes loaded successfully!").
   		Show()

   	return nil
   }

   func generateID() string {
   	return time.Now().Format("20060102150405")
   }
   ```

   **这里发生了什么：**

   - **Note 结构体**：定义了带有 JSON 标签（小写）的数据结构，以实现正确的序列化
   - **CRUD 操作**：GetAll、Create、Update 和 Delete，用于在内存中管理笔记
   - **文件对话框**：使用 `application.Get().Dialog.SaveFile()` 和 `application.Get().Dialog.OpenFile()` 访问原生对话框
   - **信息对话框**：使用 `application.Get().Dialog.Info()` 显示成功消息
   - **ID 生成**：简单的基于时间戳的 ID 生成器

3. **更新 main.go**

   替换 `main.go` 的内容：

   ```go
   package main

   import (
   	"embed"
   	_ "embed"
   	"log"

   	"github.com/wailsapp/wails/v3/pkg/application"
   )

   //go:embed all:frontend/dist

   		URL:              "/",
   	})

   	if err != nil {
   		log.Fatal(err)
   	}
   }
   ```

   **这里发生了什么：**

   - 将 `NotesService` 注册到应用程序

6. **实现 JavaScript 逻辑**

   替换 `frontend/src/main.js`：

   ```javascript
   import { NotesService } from '../bindings/changeme'

   let notes = []
   let currentNote = null

   // Load notes on startup
   async function loadNotes() {
       notes = await NotesService.GetAll()
       renderNotesList()
   }

   // Render notes list
   function renderNotesList() {
       const notesList = document.getElementById('notes-list')

       if (notes.length === 0) {
           notesList.innerHTML = '<div style="padding: 20px; text-align: center; color: #999;">No notes yet</div>'
           return
       }

       notesList.innerHTML = notes.map(note => `
           <div class="note-item ${currentNote?.id === note.id ? 'active' : ''}" data-id="${note.id}">
               <h3>${note.title || 'Untitled'}</h3>
               <p>${note.content || 'No content'}</p>
           </div>
       `).join('')

       // Add click handlers
       document.querySelectorAll('.note-item').forEach(item => {
           item.addEventListener('click', () => {
               const id = item.dataset.id
               selectNote(id)
           })
       })
   }

   // Select a note
   function selectNote(id) {
       currentNote = notes.find(n => n.id === id)
       if (currentNote) {
           document.getElementById('empty-state').style.display = 'none'
           document.getElementById('note-editor').style.display = 'flex'
           document.getElementById('note-title').value = currentNote.title
           document.getElementById('note-content').value = currentNote.content
           document.getElementById('last-updated').textContent =
               `Last updated: ${new Date(currentNote.updatedAt).toLocaleString()}`
           renderNotesList()
       }
   }

   // Create new note
   document.getElementById('new-note-btn').addEventListener('click', async () => {
       const note = await NotesService.Create('Untitled', '')
       notes.push(note)
       selectNote(note.id)
       // Focus the title input and select all text so user can immediately type
       const titleInput = document.getElementById('note-title')
       titleInput.focus()
       titleInput.select()
   })

   // Update note on input
   let updateTimeout
   function scheduleUpdate() {
       clearTimeout(updateTimeout)
       updateTimeout = setTimeout(async () => {
           if (currentNote) {
               const title = document.getElementById('note-title').value
               const content = document.getElementById('note-content').value

               await NotesService.Update(currentNote.id, title, content)

               // Update local copy
               const note = notes.find(n => n.id === currentNote.id)
               if (note) {
                   note.title = title
                   note.content = content
                   note.updatedAt = new Date().toISOString()
               }

               renderNotesList()
               document.getElementById('last-updated').textContent =
                   `Last updated: ${new Date().toLocaleString()}`
           }
       }, 500)
   }

   document.getElementById('note-title').addEventListener('input', scheduleUpdate)
   document.getElementById('note-content').addEventListener('input', scheduleUpdate)

   // Delete note
   document.getElementById('delete-btn').addEventListener('click', async () => {
       if (!currentNote) return

       try {
           await NotesService.Delete(currentNote.id)
           notes = notes.filter(n => n.id !== currentNote.id)
           currentNote = null
           document.getElementById('empty-state').style.display = 'flex'
           document.getElementById('note-editor').style.display = 'none'
           renderNotesList()
       } catch (error) {
           console.error('Delete failed:', error)
       }
   })

   // Save to file
   document.getElementById('save-btn').addEventListener('click', async () => {
       try {
           await NotesService.SaveToFile()
       } catch (error) {
           if (error) console.error('Save failed:', error)
       }
   })// 从文件加载
   document.getElementById('load-btn').addEventListener('click', async () => {
       try {
           await NotesService.LoadFromFile()
           notes = await NotesService.GetAll()
           currentNote = null
           document.getElementById('empty-state').style.display = 'flex'
           document.getElementById('note-editor').style.display = 'none'
           renderNotesList()
       } catch (error) {
           if (error) console.error('Load failed:', error)
       }
   })

   // 初始化
   loadNotes()
   ```

   **这里发生了什么：**

   - **自动保存**：500ms 防抖机制防止在输入时产生过多的后端调用
   - **属性访问**：使用小写属性名（`.id`、`.title`）以匹配 Go JSON 标签
   - **焦点管理**：创建新笔记时自动聚焦并选中标题
   - **无确认删除**：浏览器中的 `confirm()` 在 Wails WebView 中不起作用
   - **文件操作**：原生对话框处理保存/加载，并具备适当的错误处理

## 关键概念

### 包级对话框函数

在 Wails v3 中，文件和消息对话框是包级函数，而不是应用实例上的方法：

```go
// 正确 - 包级函数
path, err := application.SaveFileDialog().
    SetFilename("notes.json").
    AddFilter("JSON Files", "*.json").
    PromptForSingleSelection()

// 错误 - 不可用
path, err := app.Dialog.SaveFile() // 此方法不存在
```

### JSON 标签映射

Go 结构体的 JSON 标签必须为小写，以匹配 JavaScript 的属性访问：

```go
type Note struct {
    ID string `json:"id"` // 必须为小写
}
```

```javascript
// JavaScript 使用小写进行访问
const noteId = note.id // 正确
const noteId = note.ID // 将为 undefined
```

### 防抖自动保存

500ms 的防抖机制减少了不必要的后端调用：

```javascript
let updateTimeout
function scheduleUpdate() {
    clearTimeout(updateTimeout) // 取消之前的定时器
    updateTimeout = setTimeout(async () => {
        // 仅当用户停止输入 500ms 后才保存
        await NotesService.Update(currentNote.id, title, content)
    }, 500)
}
```
