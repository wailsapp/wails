Wails bietet eine einfache Browser-Integration über die BrowserManager-API, die es Ihrer Anwendung ermöglicht, URLs und Dateien im Standard-Webbrowser des Benutzers zu öffnen. Dies ist nützlich für das Öffnen externer Links, Dokumentation oder Dateien, die vom Browser verarbeitet werden sollen.

## Zugriff auf den Browser-Manager

Der Browser-Manager wird über die Eigenschaft `Browser` Ihrer Anwendungsinstanz angesprochen:

```go
app := application.New(application.Options{
    Name: "Browser Integration Demo",
})

// Zugriff auf den Browser-Manager
browser := app.Browser
```

## URLs öffnen

### Web-URLs öffnen

Öffnen Sie URLs im Standard-Webbrowser des Benutzers:

```go
// Eine Website öffnen
err := app.Browser.OpenURL("https://wails.io")
if err != nil {
    app.Logger.Error("Failed to open URL", "error", err)
}

// Bestimmte Seiten öffnen
err = app.Browser.OpenURL("https://github.com/wailsapp/wails")
if err != nil {
    app.Logger.Error("Failed to open GitHub", "error", err)
}
```

### Lokale URLs öffnen

Öffnen Sie lokale Entwicklungsserver oder lokale Netzwerkressourcen:

```go
// Lokalen Entwicklungsserver öffnen
err := app.Browser.OpenURL("http://localhost:3000")
if err != nil {
    app.Logger.Error("Failed to open local server", "error", err)
}

// Netzwerkressource öffnen
err = app.Browser.OpenURL("http://192.168.1.100:8080")
if err != nil {
    app.Logger.Error("Failed to open network resource", "error", err)
}
```

## Dateien öffnen

### HTML-Dateien öffnen

Öffnen Sie lokale HTML-Dateien im Browser:

```go
// Eine HTML-Datei öffnen
err := app.Browser.OpenFile("/path/to/documentation.html")
if err != nil {
    app.Logger.Error("Failed to open HTML file", "error", err)
}

// Generierte Berichte öffnen
reportPath := "/tmp/report.html"
err = app.Browser.OpenFile(reportPath)
if err != nil {
    app.Logger.Error("Failed to open report", "error", err)
}
```

### Andere Dateitypen öffnen

Öffnen Sie verschiedene Dateitypen, die Browser verarbeiten können:

```go
// PDF-Dateien öffnen
err := app.Browser.OpenFile("/path/to/document.pdf")
if err != nil {
    app.Logger.Error("Failed to open PDF", "error", err)
}

// Bilddateien öffnen
err = app.Browser.OpenFile("/path/to/image.png")
if err != nil {
    app.Logger.Error("Failed to open image", "error", err)
}

// Textdateien öffnen
err = app.Browser.OpenFile("/path/to/readme.txt")
if err != nil {
    app.Logger.Error("Failed to open text file", "error", err)
}
```

### Externe Links im Inhalt

Verwalten Sie externe Links aus dem Inhalt Ihrer Anwendung:

```go
func handleExternalLink(app *application.App, url string) {
    // URL vor dem Öffnen validieren
    if !isValidURL(url) {
        app.Logger.Warn("Invalid URL", "url", url)
        return
    }
    
    // Optional mit Benutzer bestätigen
    dialog := app.Dialog.Question()
    dialog.SetTitle("Open External Link")
    dialog.SetMessage(fmt.Sprintf("Open %s in your browser?", url))
    
    dialog.AddButton("Open").OnClick(func() {
        err := app.Browser.OpenURL(url)
        if err != nil {
            app.Logger.Error("Failed to open URL", "url", url, "error", err)
        }
    })
    
    dialog.AddButton("Cancel")
    dialog.Show()
}

func isValidURL(url string) bool {
    parsed, err := url.Parse(url)
    return err == nil && (parsed.Scheme == "http" || parsed.Scheme == "https")
}
```

### Berichte exportieren und anzeigen

Generieren und öffnen Sie Berichte im Browser:

```go
import (
    "html/template"
    "os"
    "path/filepath"
)

func generateAndOpenReport(app *application.App, data interface{}) error {
    // Temporäre Datei für den Bericht erstellen
    tmpDir := os.TempDir()
    reportPath := filepath.Join(tmpDir, "report.html")
    
    // HTML-Bericht generieren
    tmpl := `
<!DOCTYPE html>
<html>
<head>
    <title>Application Report</title>
    <style>
        body { font-family: Arial, sans-serif; margin: 40px; }
        .header { border-bottom: 2px solid #333; padding-bottom: 10px; }
        .data { margin-top: 20px; }
    </style>
</head>
<body>
    <div class="header">
        <h1>Application Report</h1>
        <p>Generated: {{.Timestamp}}</p>
    </div>
    <div class="data">
        <!-- Report content here -->
        {{range .Items}}
        <p>{{.}}</p>
        {{end}}
    </div>
</body>
</html>`
    
    // Bericht in Datei schreiben
    file, err := os.Create(reportPath)
    if err != nil {
        return err
    }
    defer file.Close()
    
    t, err := template.New("report").Parse(tmpl)
    if err != nil {
        return err
    }
    
    err = t.Execute(file, data)
    if err != nil {
        return err
    }
    
    // Im Browser öffnen
    return app.Browser.OpenFile(reportPath)
}
```

## Fehlerbehandlung

### Elegante Fehlerbehandlung

Behandeln Sie stets potenzielle Fehler beim Öffnen von URLs oder Dateien:

```go
func openURLWithFallback(app *application.App, url string, fallbackMessage string) {
    err := app.Browser.OpenURL(url)
    if err != nil {
        app.Logger.Error("Failed to open URL", "url", url, "error", err)
        
        // Fallback-Dialog mit URL anzeigen
        dialog := app.Dialog.Info()
        dialog.SetTitle("Unable to Open Link")
        dialog.SetMessage(fmt.Sprintf("%s\n\nURL: %s", fallbackMessage, url))
        dialog.Show()
    }
}

// Verwendung
openURLWithFallback(app, 
    "https://docs.example.com", 
    "Please open the following URL manually in your browser:")
```

### Benutzerfeedback

Geben Sie Feedback, wenn Vorgänge erfolgreich oder fehlgeschlagen sind:

```go
func openURLWithFeedback(app *application.App, url string) {
    err := app.Browser.OpenURL(url)
    if err != nil {
        // Fehlerdialog anzeigen
        app.Dialog.Error().
            SetTitle("Browser Error").
            SetMessage(fmt.Sprintf("Could not open URL: %s", err.Error())).
            Show()
    } else {
        // Optional Erfolgsmeldung anzeigen
        app.Logger.Info("URL opened successfully", "url", url)
    }
}
```

## Plattform-spezifische Hinweise

<Tabs>
  <TabItem label="macOS" icon="fa-brands:apple">
    
    Unter macOS:
    
    - Verwendet den Befehl `open`, um den Standardbrowser zu starten- Respektiert die Standardeinstellung des Benutzers für den Webbrowser in den Systemeinstellungen
- Kann bei sandgeboxten Anwendungen nach Berechtigungen fragen
- Verarbeitet `file://`-URLs für lokale Dateien korrekt
  
  </TabItem>
  
  <TabItem label="Windows" icon="fa-brands:windows">
    
    Unter Windows:
    
    - Verwendet die Windows Shell API, um URLs zu öffnen
    - Respektiert die Standardeinstellung des Webbrowsers in den Windows-Einstellungen
    - Verarbeitet Windows-Pfadformate korrekt
    - Kann Sicherheitswarnungen für nicht vertrauenswürdige URLs anzeigen
    
  </TabItem>
  
  <TabItem label="Linux" icon="fa-brands:linux">
    
    Unter Linux:
    
    - Versucht zunächst `xdg-open` zu verwenden, fällt auf andere Methoden zurück
    - Das Verhalten variiert je nach Desktop-Umgebung
    - Respektiert die Umgebungsvariable `BROWSER`, falls gesetzt
    - Kann zusätzliche Pakete bei minimalen Installationen erfordern
    
  </TabItem>
</Tabs>

## Best Practices

1. **Fehler immer behandeln**: Browseroperationen können aus verschiedenen Gründen fehlschlagen:
   ```go
   if err := app.Browser.OpenURL(url); err != nil {
       app.Logger.Error("Failed to open browser", "error", err)
       // Provide fallback or user notification
   }
   ```

2. **URLs validieren**: Stellen Sie sicher, dass URLs vor dem Öffnen wohlgeformt sind:
   ```go
   func isValidHTTPURL(str string) bool {
       u, err := url.Parse(str)
       return err == nil && (u.Scheme == "http" || u.Scheme == "https")
   }
   ```

3. **Benutzerbestätigung**: Für externe Links sollten Sie die Erlaubnis des Benutzers einholen:
   ```go
   // Show confirmation dialog before opening external links
   confirmAndOpen(app, "https://external-site.com")
   ```

4. **Sichere Dateipfade**: Beim Öffnen von Dateien stellen Sie sicher, dass die Pfade sicher sind:
   ```go
   func openSafeFile(app *application.App, filename string) error {
       // Ensure file exists and is readable
       if _, err := os.Stat(filename); err != nil {
           return err
       }
       return app.Browser.OpenFile(filename)
   }
   ```

## Vollständiges Beispiel

Hier ist ein vollständiges Beispiel, das verschiedene Muster zur Browserintegration zeigt:

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "github.com/wailsapp/wails/v3/pkg/application"
)

func main() {
    app := application.New(application.Options{
        Name: "Browser Integration Demo",
    })

    // Setup menu with browser actions
    setupMenu(app)

    // Create main window
    window := app.Window.New()
    window.SetTitle("Browser Integration")

    err := app.Run()
    if err != nil {
        panic(err)
    }
}

func setupMenu(app *application.App) {
    menu := app.Menu.New()
    
    // File menu
    fileMenu := menu.AddSubmenu("File")
    fileMenu.Add("Generate Report").OnClick(func(ctx *application.Context) {
        generateHTMLReport(app)
    })
    
    // Help menu
    helpMenu := menu.AddSubmenu("Help")
    helpMenu.Add("Documentation").OnClick(func(ctx *application.Context) {
        openWithConfirmation(app, "https://docs.example.com")
    })
    helpMenu.Add("Support").OnClick(func(ctx *application.Context) {
        openWithConfirmation(app, "https://support.example.com")
    })
    
    app.Menu.Set(menu)
}

func openWithConfirmation(app *application.App, url string) {
    dialog := app.Dialog.Question()
    dialog.SetTitle("Open External Link")
    dialog.SetMessage(fmt.Sprintf("Open %s in your browser?", url))
    
    dialog.AddButton("Open").OnClick(func() {
        if err := app.Browser.OpenURL(url); err != nil {
            showError(app, "Failed to open URL", err)
        }
    })
    
    dialog.AddButton("Cancel")
    dialog.Show()
}

func generateHTMLReport(app *application.App) {
    // Create temporary HTML file
    tmpDir := os.TempDir()
    reportPath := filepath.Join(tmpDir, "demo_report.html")
    
    html := `
<!DOCTYPE html>
<html>
<head>
    <title>Demo Report</title>
    <style>
        body { font-family: Arial, sans-serif; margin: 40px; }
        .header { color: #333; border-bottom: 1px solid #ccc; }
    </style>
</head>
<body>
    <div class="header">
        <h1>Application Report</h1>
        <p>This is a sample report generated by the application.</p>
    </div>
    <div class="content">
        <h2>Report Details</h2>
        <p>This report was generated to demonstrate browser integration.</p>
    </div>
</body>
</html>`
    
    err := os.WriteFile(reportPath, []byte(html), 0644)
    if err != nil {
        showError(app, "Failed to create report", err)
        return
    }
    
    // Open in browser
    err = app.Browser.OpenFile(reportPath)
    if err != nil {
        showError(app, "Failed to open report", err)
    }
}

func showError(app *application.App, message string, err error) {
    app.Dialog.Error().
        SetTitle("Error").
        SetMessage(fmt.Sprintf("%s: %v", message, err)).
        Show()
}
```

:::tip[Pro Tip]
Consider providing fallback mechanisms for when browser operations fail, such as copying URLs to clipboard or showing them in a dialog for manual opening.
:::

:::danger[Warning]
Always validate URLs and file paths before opening them to prevent security issues. Be cautious about opening user-provided URLs without validation.
:::
