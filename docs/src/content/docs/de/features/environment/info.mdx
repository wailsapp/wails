Wails stellt umfassende Umgebungsdaten über die EnvironmentManager-API bereit. Dies ermöglicht es Ihrer Anwendung, Systemeigenschaften, Themenpräferenzen zu erkennen und sich in den Dateimanager des Betriebssystems zu integrieren.

## Zugriff auf den Environment Manager

Der Environment Manager wird über die `Env`-Eigenschaft Ihrer Anwendungsinstanz angesprochen:

```go
app := application.New(application.Options{
    Name: "Environment Demo",
})

// Greifen Sie auf den Environment Manager zu
env := app.Env
```

## Systeminformationen

### Umgebungsdaten abrufen

Rufen Sie umfassende Informationen über die Laufzeitumgebung ab:

```go
envInfo := app.Env.Info()

app.Logger.Info("Umgebungsdaten",
    "os", envInfo.OS,           // "windows", "darwin", "linux"
    "arch", envInfo.Arch,       // "amd64", "arm64", usw.
    "debug", envInfo.Debug,     // Debug-Modus-Flag
)

// Details zum Betriebssystem
if envInfo.OSInfo != nil {
    app.Logger.Info("OS-Details",
        "name", envInfo.OSInfo.Name,
        "version", envInfo.OSInfo.Version,
    )
}

// Plattform-spezifische Informationen
for key, value := range envInfo.PlatformInfo {
    app.Logger.Info("Plattform-Info", "key", key, "value", value)
}
```

### Umgebungsstruktur

Die Umgebungsdaten enthalten mehrere wichtige Felder:

```go
type EnvironmentInfo struct {
    OS           string                 // Betriebssystem: "windows", "darwin", "linux"
    Arch         string                 // Architektur: "amd64", "arm64", "386", usw.
    Debug        bool                   // Ob im Debug-Modus ausgeführt wird
    OSInfo       *operatingsystem.OS    // Detaillierte OS-Informationen
    PlatformInfo map[string]any         // Plattform-spezifische Details
}
```

## Themen-Erkennung

### Dark Mode-Erkennung

Erkennen Sie, ob das System den Dark Mode verwendet:

```go
if app.Env.IsDarkMode() {
    app.Logger.Info("Das System ist im Dark Mode")
    // Wenden Sie das dunkle Thema auf Ihre Anwendung an
    applyDarkTheme()
} else {
    app.Logger.Info("Das System ist im Light Mode")
    // Wenden Sie das helle Thema auf Ihre Anwendung an
    applyLightTheme()
}
```

### Themenänderungen überwachen

Hören Sie auf Themenänderungen, um Ihre Anwendung dynamisch zu aktualisieren:

```go
import "github.com/wailsapp/wails/v3/pkg/events"

// Hören Sie auf Themenänderungen
app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
    if app.Env.IsDarkMode() {
        app.Logger.Info("Zum Dark Mode gewechselt")
        updateApplicationTheme("dark")
    } else {
        app.Logger.Info("Zum Light Mode gewechselt")
        updateApplicationTheme("light")
    }
})

func updateApplicationTheme(theme string) {
    // Aktualisieren Sie das Thema Ihrer Anwendung
    // Dies könnte ein Event an das Frontend senden
    app.Event.Emit("theme:changed", theme)
}
```

## Dateimanager-Integration

### Dateimanager öffnen

Öffnen Sie den Dateimanager des Systems an einem bestimmten Speicherort:

```go
// Öffnen Sie den Dateimanager in einem Verzeichnis
err := app.Env.OpenFileManager("/Users/username/Documents", false)
if err != nil {
    app.Logger.Error("Dateimanager konnte nicht geöffnet werden", "error", err)
}

// Öffnen Sie den Dateimanager und markieren Sie eine bestimmte Datei
err = app.Env.OpenFileManager("/Users/username/Documents/report.pdf", true)
if err != nil {
    app.Logger.Error("Dateimanager mit Markierung konnte nicht geöffnet werden", "error", err)
}
```

### Häufige Anwendungsfälle

Zeigen Sie Dateien oder Ordner im Dateimanager von Ihrer Anwendung aus an:

```go
func showInFileManager(app *application.App, path string) {
    err := app.Env.OpenFileManager(path, true)
    if err != nil {
        // Fallback: Versuchen Sie, nur das Verzeichnis zu öffnen
        dir := filepath.Dir(path)
        err = app.Env.OpenFileManager(dir, false)
        if err != nil {
            app.Logger.Error("Dateimanager konnte nicht geöffnet werden", "path", path, "error", err)
            
            // Zeigen Sie den Fehler dem Benutzer an
            app.Dialog.Error().
                SetTitle("Dateimanager-Fehler").
                SetMessage("Dateimanager konnte nicht geöffnet werden").
                Show()
        }
    }
}

// Nutzungsbeispiele
func setupFileMenu(app *application.App) {
    menu := app.Menu.New()
    fileMenu := menu.AddSubmenu("Datei")
    
    fileMenu.Add("Downloads-Ordner anzeigen").OnClick(func(ctx *application.Context) {
        homeDir, _ := os.UserHomeDir()
        downloadsDir := filepath.Join(homeDir, "Downloads")
        showInFileManager(app, downloadsDir)
    })
    
    fileMenu.Add("Anwendungsdaten anzeigen").OnClick(func(ctx *application.Context) {
        configDir, _ := os.UserConfigDir()
        appDir := filepath.Join(configDir, "MyApp")
        showInFileManager(app, appDir)
    })
}
```

## Plattform-spezifisches Verhalten

### Adaptive Anwendungslogik

Verwenden Sie Umgebungsdaten, um das Verhalten Ihrer Anwendung anzupassen:

```go
func configureForPlatform(app *application.App) {
    envInfo := app.Env.Info()
    
    switch envInfo.OS {
    case "darwin":
        configureMacOS(app)
    case "windows":
        configureWindows(app)
    case "linux":
        configureLinux(app)
    }
    
    // An die Architektur anpassen
    if envInfo.Arch == "arm64" {
        app.Logger.Info("Wird auf ARM-Architektur ausgeführt")
        // Möglicherweise für ARM optimieren
    }
}

func configureMacOS(app *application.App) {
    app.Logger.Info("Konfiguration für macOS")
    
    // macOS-spezifische Konfiguration
    menu := app.Menu.New()
    menu.AddRole(application.AppMenu) // Standard-macOS-Anwendungsmenü hinzufügen
    
    // Dark Mode behandeln
    if app.Env.IsDarkMode() {
        setMacOSDarkTheme()
    }
}

func configureWindows(app *application.App) {
    app.Logger.Info("Konfiguration für Windows")
    
    // Windows-spezifische Konfiguration
    // Windows-Stil-Menüs, Tastenkombinationen usw. einrichten
}

func configureLinux(app *application.App) {
    app.Logger.Info("Konfiguration für Linux")
    
    // Linux-spezifische Konfiguration
    // An verschiedene Desktop-Umgebungen anpassen
}
```

## Behandlung des Debug-Modus

### Entwicklung vs. Produktion

Verwenden Sie die Debug-Modus-Informationen, um Entwicklungsfunktionen zu aktivieren:

```go
func setupApplicationMode(app *application.App) {
    envInfo := app.Env.Info()
    
    if envInfo.Debug {
        app.Logger.Info("Wird im Debug-Modus ausgeführt")
        setupDevelopmentFeatures(app)
    } else {
        app.Logger.Info("Wird im Produktionsmodus ausgeführt")
        setupProductionFeatures(app)
    }
}

func setupDevelopmentFeatures(app *application.App) {
    // Nur für die Entwicklung verfügbare Funktionen aktivieren
    menu := app.Menu.New()
    
    // Entwicklungsmenü hinzufügen
    devMenu := menu.AddSubmenu("Entwicklung")
    devMenu.Add("Anwendung neu laden").OnClick(func(ctx *application.Context) {
        // Anwendung neu laden
        window := app.Window.Current()
        if window != nil {
            window.Reload()
        }
    })
    
    devMenu.Add("DevTools öffnen").OnClick(func(ctx *application.Context) {
        window := app.Window.Current()
        if window != nil {
            window.OpenDevTools()
        }
    })
    
    devMenu.Add("Umgebung anzeigen").OnClick(func(ctx *application.Context) {
        showEnvironmentDialog(app)
    })
}

func setupProductionFeatures(app *application.App) {
    // Nur für die Produktion verfügbare Funktionen
    // Debug-Logging deaktivieren, Analysen aktivieren, usw.
}
```

## Dialog für Umgebungsdaten

### Systeminformationen anzeigen

Erstellen Sie einen Dialog, der Umgebungsdaten anzeigt:

```go
func showEnvironmentDialog(app *application.App) {
    envInfo := app.Env.Info()
    
    details := fmt.Sprintf(`Umgebungsdaten:

Betriebssystem: %s
Architektur: %s
Debug-Modus: %t

Dark Mode: %t

Plattform-Informationen:`, 
        envInfo.OS, 
        envInfo.Arch, 
        envInfo.Debug,
        app.Env.IsDarkMode())
    
    // Plattform-spezifische Details hinzufügen
    for key, value := range envInfo.PlatformInfo {
        details += fmt.Sprintf("\n%s: %v", key, value)
    }if envInfo.OSInfo != nil {
        details += fmt.Sprintf("\n\nOS Details:\nName: %s\nVersion: %s", 
            envInfo.OSInfo.Name, 
            envInfo.OSInfo.Version)
    }
    
    dialog := app.Dialog.Info()
    dialog.SetTitle("Environment Information")
    dialog.SetMessage(details)
    dialog.Show()
}
```

## Platform considerations

<Tabs>
  <TabItem label="macOS" icon="fa-brands:apple">
    
    Unter macOS:
    
    - Die Erkennung des Dark Mode verwendet die Systemeinstellungen für das Erscheinungsbild
    - Datei-Manager-Operationen verwenden Finder
    - Die Plattforminformationen enthalten Details zur macOS-Version
    - Die Architektur kann auf Apple-Silicon-Macs „arm64" sein
    
    ```go
    if envInfo.OS == "darwin" {
        // macOS-spezifische Verarbeitung
        if envInfo.Arch == "arm64" {
            app.Logger.Info("Running on Apple Silicon")
        }
    }
    ```
    
  </TabItem>
  
  <TabItem label="Windows" icon="fa-brands:windows">
    
    Unter Windows:
    
    - Die Erkennung des Dark Mode verwendet die Windows-Theme-Einstellungen
    - Datei-Manager-Operationen verwenden den Windows-Explorer
    - Die Plattforminformationen enthalten Details zur Windows-Version
    - Können zusätzliche, windows-spezifische Informationen enthalten
    
    ```go
    if envInfo.OS == "windows" {
        // Windows-spezifische Verarbeitung
        for key, value := range envInfo.PlatformInfo {
            if key == "windows_version" {
                app.Logger.Info("Windows version", "version", value)
            }
        }
    }
    ```
    
  </TabItem>
  
  <TabItem label="Linux" icon="fa-brands:linux">
    
    Unter Linux:
    
    - Die Erkennung des Dark Mode variiert je nach Desktop-Umgebung
    - Datei-Manager-Operationen verwenden den standardmäßigen Datei-Manager des Systems
    - Die Plattforminformationen enthalten Details zur Distribution
    - Das Verhalten kann zwischen verschiedenen Linux-Distributionen variieren
    
    ```go
    if envInfo.OS == "linux" {
        // Linux-spezifische Verarbeitung
        if distro, ok := envInfo.PlatformInfo["distribution"]; ok {
            app.Logger.Info("Linux distribution", "distro", distro)
        }
    }
    ```

## Best Practices

1. **Umgebungsinformationen zwischenspeichern**: Umgebungsinformationen ändern sich während der Laufzeit selten:
   ```go
   type App struct {
       envInfo *application.EnvironmentInfo
   }
   
   func (a *App) getEnvInfo() application.EnvironmentInfo {
       if a.envInfo == nil {
           info := a.app.Env.Info()
           a.envInfo = &info
       }
       return *a.envInfo
   }
   ```

2. **Theme-Änderungen verarbeiten**: Auf System-Theme-Änderungen hören:
   ```go
   app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
       updateTheme(app.Env.IsDarkMode())
   })
   ```

3. **Fehler beim Datei-Manager graceful behandeln**: Datei-Manager-Fehler immer behandeln:
   ```go
   func openFileManagerSafely(app *application.App, path string) {
       err := app.Env.OpenFileManager(path, false)
       if err != nil {
           // Fallback oder Benutzernachricht bereitstellen
           app.Logger.Warn("Could not open file manager", "path", path)
       }
   }
   ```

4. **Plattform-spezifische Funktionen**: Umgebungsinformationen nutzen, um plattformspezifische Funktionen zu aktivieren:
   ```go
   envInfo := app.Env.Info()
   if envInfo.OS == "darwin" {
       // macOS-spezifische Funktionen aktivieren
   }
   ```

## Vollständiges Beispiel

Hier ist ein vollständiges Beispiel, das das Umgebungsmanagement demonstriert:

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "runtime"
    "github.com/wailsapp/wails/v3/pkg/application"
    "github.com/wailsapp/wails/v3/pkg/events"
)

func main() {
    app := application.New(application.Options{
        Name: "Environment Demo",
    })

    // Anwendung basierend auf der Umgebung einrichten
    setupForEnvironment(app)

    // Theme-Änderungen überwachen
    monitorThemeChanges(app)

    // Menü mit Umgebungsfunktionen erstellen
    setupEnvironmentMenu(app)

    // Hauptfenster erstellen
    window := app.Window.New()
    window.SetTitle("Environment Demo")

    err := app.Run()
    if err != nil {
        panic(err)
    }
}

func setupForEnvironment(app *application.App) {
    envInfo := app.Env.Info()
    
    app.Logger.Info("Application environment",
        "os", envInfo.OS,
        "arch", envInfo.Arch,
        "debug", envInfo.Debug,
        "darkMode", app.Env.IsDarkMode(),
    )

    // Für Plattform konfigurieren
    switch envInfo.OS {
    case "darwin":
        app.Logger.Info("Configuring for macOS")
        // macOS-spezifische Einrichtung
    case "windows":
        app.Logger.Info("Configuring for Windows")
        // Windows-spezifische Einrichtung
    case "linux":
        app.Logger.Info("Configuring for Linux")
        // Linux-spezifische Einrichtung
    }

    // Initiales Theme anwenden
    if app.Env.IsDarkMode() {
        applyDarkTheme(app)
    } else {
        applyLightTheme(app)
    }
}

func monitorThemeChanges(app *application.App) {
    app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
        if app.Env.IsDarkMode() {
            app.Logger.Info("System switched to dark mode")
            applyDarkTheme(app)
        } else {
            app.Logger.Info("System switched to light mode")
            applyLightTheme(app)
        }
    })
}

func setupEnvironmentMenu(app *application.App) {
    menu := app.Menu.New()
    
    // Plattformspezifisches App-Menü hinzufügen
    if runtime.GOOS == "darwin" {
        menu.AddRole(application.AppMenu)
    }
    
    // Tools-Menü
    toolsMenu := menu.AddSubmenu("Tools")
    
    toolsMenu.Add("Show Environment Info").OnClick(func(ctx *application.Context) {
        showEnvironmentInfo(app)
    })
    
    toolsMenu.Add("Open Downloads Folder").OnClick(func(ctx *application.Context) {
        openDownloadsFolder(app)
    })
    
    toolsMenu.Add("Toggle Theme").OnClick(func(ctx *application.Context) {
        // Dies würde typischerweise vom System behandelt
        // wird hier jedoch zur Demonstration gezeigt
        toggleTheme(app)
    })
    
    app.Menu.Set(menu)
}

func showEnvironmentInfo(app *application.App) {
    envInfo := app.Env.Info()
    
    message := fmt.Sprintf(`Environment Information:

Operating System: %s
Architecture: %s
Debug Mode: %t
Dark Mode: %t

Platform Details:`,
        envInfo.OS,
        envInfo.Arch,
        envInfo.Debug,
        app.Env.IsDarkMode())
    
    for key, value := range envInfo.PlatformInfo {
        message += fmt.Sprintf("\n%s: %v", key, value)
    }
    
    if envInfo.OSInfo != nil {
        message += fmt.Sprintf("\n\nOS Information:\nName: %s\nVersion: %s",
            envInfo.OSInfo.Name,
            envInfo.OSInfo.Version)
    }
    
    dialog := app.Dialog.Info()
    dialog.SetTitle("Environment Information")
    dialog.SetMessage(message)
    dialog.Show()
}

func openDownloadsFolder(app *application.App) {
    homeDir, err := os.UserHomeDir()
    if err != nil {
        app.Logger.Error("Could not get home directory", "error", err)
        return
    }
    
    downloadsDir := filepath.Join(homeDir, "Downloads")
    err = app.Env.OpenFileManager(downloadsDir, false)
    if err != nil {
        app.Logger.Error("Could not open Downloads folder", "error", err)
        
        app.Dialog.Error().
            SetTitle("File Manager Error").
            SetMessage("Could not open Downloads folder").
            Show()
    }
}

func applyDarkTheme(app *application.App) {
    app.Logger.Info("Applying dark theme")
    // Theme-Änderung an Frontend senden
    app.Event.Emit("theme:apply", "dark")
}

func applyLightTheme(app *application.App) {
    app.Logger.Info("Applying light theme")
    // Theme-Änderung an Frontend senden
    app.Event.Emit("theme:apply", "light")
}

func toggleTheme(app *application.App) {
    // Dies dient nur der Demonstration
    // Echte Theme-Änderungen sollten vom System kommen
    currentlyDark := app.Env.IsDarkMode()
    if currentlyDark {
        applyLightTheme(app)
    } else {
        applyDarkTheme(app)
    }
}
```

:::tip[Tipp]
Nutzen Sie die Umgebungsinformationen, um plattformgerechte Benutzererfahrungen zu bieten. Verwenden Sie beispielsweise die Command-Taste für Shortcuts auf macOS und die Control-Taste für Shortcuts auf Windows/Linux.
:::

:::danger[Achtung]
Die Umgebungsinformationen sind während der Laufzeit der Anwendung im Allgemeinen stabil, aber die Themenpräferenzen können sich ändern. Hören Sie immer auf Themenänderungsereignisse, um Ihre Benutzeroberfläche synchron zu halten.
:::
