<script>{`
  (function() {
    const words = ['desktop', 'server'];
    const transitions = ['fade-out', 'blur-out', 'scale-out', 'slide-out'];
    let currentIndex = 0;

    function init() {
      // Find the tagline and replace 'desktop' with morphing element
      const tagline = document.querySelector('.hero > div > p, .sl-markdown-content p:first-of-type, [class*="tagline"]');
      if (!tagline) {
        // Retry after a short delay if not found
        setTimeout(init, 500);
        return;
      }

      const html = tagline.innerHTML;
      if (html.includes('morph-word')) return; // Already initialized

      const newHtml = html.replace(
        /desktop/i,
        '<span class="morph-word"><span>desktop</span></span>'
      );
      tagline.innerHTML = newHtml;

      // Start the morphing cycle
      scheduleNext();
    }

    function scheduleNext() {
      // Random interval between 3-5 seconds
      const delay = 3000 + Math.random() * 2000;
      setTimeout(morph, delay);
    }

    function morph() {
      const morphSpan = document.querySelector('.morph-word span');
      if (!morphSpan) return;

      // Pick random transition
      const transition = transitions[Math.floor(Math.random() * transitions.length)];

      // Fade out with transition
      morphSpan.classList.add(transition);

      // After transition, swap text and fade in
      setTimeout(() => {
        currentIndex = (currentIndex + 1) % words.length;
        morphSpan.textContent = words[currentIndex];
        morphSpan.classList.remove(transition);

        // Schedule next morph
        scheduleNext();
      }, 800);
    }

    // Initialize when DOM is ready
    if (document.readyState === 'loading') {
      document.addEventListener('DOMContentLoaded', init);
    } else {
      init();
    }
  })();
`}</script>

## Quickstart

```bash
# 1. Install the Wails CLI
go install github.com/wailsapp/wails/v3/cmd/wails3@latest

# 2. Run the setup wizard (experimental)
wails3 setup
```

Then you're ready for your first project:

```bash
wails3 init -n myapp -t vanilla
cd myapp && wails3 dev
```

**Your application is now running** with hot reload and type-safe Go-to-JS bindings.
