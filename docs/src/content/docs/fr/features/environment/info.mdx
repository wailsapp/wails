Wails fournit des informations complètes sur l'environnement via l'API EnvironmentManager. Cela permet à votre application de détecter les propriétés du système, les préférences de thème et de s'intégrer au gestionnaire de fichiers du système d'exploitation.

## Accéder au Gestionnaire d'Environnement

Le gestionnaire d'environnement est accessible via la propriété `Env` de votre instance d'application :

```go
app := application.New(application.Options{
    Name: "Environment Demo",
})

// Accéder au gestionnaire d'environnement
env := app.Env
```

## Informations Système

### Obtenir les Informations d'Environnement

Récupérez des informations complètes sur l'environnement d'exécution :

```go
envInfo := app.Env.Info()

app.Logger.Info("Environment information",
    "os", envInfo.OS,           // "windows", "darwin", "linux"
    "arch", envInfo.Arch,       // "amd64", "arm64", etc.
    "debug", envInfo.Debug,     // Indicateur du mode debug
)

// Détails du système d'exploitation
if envInfo.OSInfo != nil {
    app.Logger.Info("OS details",
        "name", envInfo.OSInfo.Name,
        "version", envInfo.OSInfo.Version,
    )
}

// Informations spécifiques à la plateforme
for key, value := range envInfo.PlatformInfo {
    app.Logger.Info("Platform info", "key", key, "value", value)
}
```

### Structure de l'Environnement

Les informations d'environnement incluent plusieurs champs importants :

```go
type EnvironmentInfo struct {
    OS           string                 // Système d'exploitation : "windows", "darwin", "linux"
    Arch         string                 // Architecture : "amd64", "arm64", "386", etc.
    Debug        bool                   // Indique si le mode debug est actif
    OSInfo       *operatingsystem.OS    // Informations détaillées sur le système d'exploitation
    PlatformInfo map[string]any         // Détails spécifiques à la plateforme
}
```

## Détection du Thème

### Détection du Mode Sombre

Détectez si le système utilise le mode sombre :

```go
if app.Env.IsDarkMode() {
    app.Logger.Info("System is in dark mode")
    // Appliquer le thème sombre à votre application
    applyDarkTheme()
} else {
    app.Logger.Info("System is in light mode")
    // Appliquer le thème clair à votre application
    applyLightTheme()
}
```

### Surveillance des Changements de Thème

Écoutez les changements de thème pour mettre à jour votre application dynamiquement :

```go
import "github.com/wailsapp/wails/v3/pkg/events"

// Écouter les changements de thème
app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
    if app.Env.IsDarkMode() {
        app.Logger.Info("Switched to dark mode")
        updateApplicationTheme("dark")
    } else {
        app.Logger.Info("Switched to light mode")
        updateApplicationTheme("light")
    }
})

func updateApplicationTheme(theme string) {
    // Mettre à jour le thème de votre application
    // Cela peut émettre un événement vers le frontend
    app.Event.Emit("theme:changed", theme)
}
```

## Intégration avec le Gestionnaire de Fichiers

### Ouvrir le Gestionnaire de Fichiers

Ouvrez le gestionnaire de fichiers du système à un emplacement spécifique :

```go
// Ouvrir le gestionnaire de fichiers à un répertoire
err := app.Env.OpenFileManager("/Users/username/Documents", false)
if err != nil {
    app.Logger.Error("Failed to open file manager", "error", err)
}

// Ouvrir le gestionnaire de fichiers et sélectionner un fichier spécifique
err = app.Env.OpenFileManager("/Users/username/Documents/report.pdf", true)
if err != nil {
    app.Logger.Error("Failed to open file manager with selection", "error", err)
}
```

### Cas d'Usage Courants

Afficher des fichiers ou des dossiers dans le gestionnaire de fichiers depuis votre application :

```go
func showInFileManager(app *application.App, path string) {
    err := app.Env.OpenFileManager(path, true)
    if err != nil {
        // Solution de repli : essayer d'ouvrir uniquement le répertoire
        dir := filepath.Dir(path)
        err = app.Env.OpenFileManager(dir, false)
        if err != nil {
            app.Logger.Error("Could not open file manager", "path", path, "error", err)
            
            // Afficher l'erreur à l'utilisateur
            app.Dialog.Error().
                SetTitle("File Manager Error").
                SetMessage("Could not open file manager").
                Show()
        }
    }
}

// Exemples d'utilisation
func setupFileMenu(app *application.App) {
    menu := app.Menu.New()
    fileMenu := menu.AddSubmenu("File")
    
    fileMenu.Add("Show Downloads Folder").OnClick(func(ctx *application.Context) {
        homeDir, _ := os.UserHomeDir()
        downloadsDir := filepath.Join(homeDir, "Downloads")
        showInFileManager(app, downloadsDir)
    })
    
    fileMenu.Add("Show Application Data").OnClick(func(ctx *application.Context) {
        configDir, _ := os.UserConfigDir()
        appDir := filepath.Join(configDir, "MyApp")
        showInFileManager(app, appDir)
    })
}
```

## Comportement Spécifique à la Plateforme

Utilisez les informations de l'environnement pour adapter le comportement de votre application :

```go
func configureForPlatform(app *application.App) {
    envInfo := app.Env.Info()
    
    switch envInfo.OS {
    case "darwin":
        configureMacOS(app)
    case "windows":
        configureWindows(app)
    case "linux":
        configureLinux(app)
    }
    
    // S'adapter à l'architecture
    if envInfo.Arch == "arm64" {
        app.Logger.Info("Running on ARM architecture")
        // Optimiser potentiellement pour ARM
    }
}

func configureMacOS(app *application.App) {
    app.Logger.Info("Configuring for macOS")
    
    // Configuration spécifique à macOS
    menu := app.Menu.New()
    menu.AddRole(application.AppMenu) // Ajouter le menu d'application standard macOS
    
    // Gérer le mode sombre
    if app.Env.IsDarkMode() {
        setMacOSDarkTheme()
    }
}

func configureWindows(app *application.App) {
    app.Logger.Info("Configuring for Windows")
    
    // Configuration spécifique à Windows
    // Mettre en place des menus et des raccourcis clavier de style Windows, etc.

    app.Logger.Info("Configuring for Linux")
    
    // Configuration spécifique à Linux
    // S'adapter aux différents environnements de bureau

## Gestion du Mode Debug

### Développement vs Production

Utilisez les informations du mode debug pour activer les fonctionnalités de développement :

```go
func setupApplicationMode(app *application.App) {
    envInfo := app.Env.Info()
    
    if envInfo.Debug {
        app.Logger.Info("Running in debug mode")
        setupDevelopmentFeatures(app)
    } else {
        app.Logger.Info("Running in production mode")
        setupProductionFeatures(app)
    }
}

func setupDevelopmentFeatures(app *application.App) {
    // Activer les fonctionnalités exclusives au développement
    menu := app.Menu.New()
    
    // Ajouter un menu de développement
    devMenu := menu.AddSubmenu("Development")
    devMenu.Add("Reload Application").OnClick(func(ctx *application.Context) {
        // Recharger l'application
        window := app.Window.Current()
        if window != nil {
            window.Reload()
        }
    })
    
    devMenu.Add("Open DevTools").OnClick(func(ctx *application.Context) {
        window := app.Window.Current()
        if window != nil {
            window.OpenDevTools()
        }
    })
    
    devMenu.Add("Show Environment").OnClick(func(ctx *application.Context) {
        showEnvironmentDialog(app)
    })
}

func setupProductionFeatures(app *application.App) {
    // Fonctionnalités exclusives à la production
    // Désactiver les logs de debug, activer l'analytics, etc.
}
```

## Boîte de Dialogue des Informations d'Environnement

### Afficher les Informations Système

Créer une boîte de dialogue affichant les informations de l'environnement :

```go
func showEnvironmentDialog(app *application.App) {
    envInfo := app.Env.Info()
    
    details := fmt.Sprintf(`Environment Information:

Operating System: %s
Architecture: %s
Debug Mode: %t

Dark Mode: %t

Platform Information:`, 
        envInfo.OS, 
        envInfo.Arch, 
        envInfo.Debug,
        app.Env.IsDarkMode())
    
    // Ajouter les détails spécifiques à la plateforme
    for key, value := range envInfo.PlatformInfo {
        details += fmt.Sprintf("\n%s: %v", key, value)
    }    if envInfo.OSInfo != nil {
        details += fmt.Sprintf("\n\nDétails du système d'exploitation :\nNom : %s\nVersion : %s", 
            envInfo.OSInfo.Name, 
            envInfo.OSInfo.Version)
    }
    
    dialog := app.Dialog.Info()
    dialog.SetTitle("Informations sur l'environnement")
    dialog.SetMessage(details)
    dialog.Show()
}
```

## Considérations relatives à la plateforme

## Bonnes pratiques

1. **Mettre en cache les informations de l'environnement** : Les informations de l'environnement changent rarement pendant l'exécution :
   ```go
   type App struct {
       envInfo *application.EnvironmentInfo
   }
   
   func (a *App) getEnvInfo() application.EnvironmentInfo {
       if a.envInfo == nil {
           info := a.app.Env.Info()
           a.envInfo = &info
       }
       return *a.envInfo
   }
   ```

2. **Gérer les changements de thème** : Écouter les changements de thème du système :
   ```go
   app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
       updateTheme(app.Env.IsDarkMode())
   })
   ```

3. **Gérer gracieusement les échecs du gestionnaire de fichiers** : Toujours gérer les erreurs du gestionnaire de fichiers :
   ```go
   func openFileManagerSafely(app *application.App, path string) {
       err := app.Env.OpenFileManager(path, false)
       if err != nil {
           // Prévoir une solution de repli ou une notification utilisateur
           app.Logger.Warn("Impossible d'ouvrir le gestionnaire de fichiers", "path", path)
       }
   }
   ```

4. **Fonctionnalités spécifiques à la plateforme** : Utiliser les informations de l'environnement pour activer les fonctionnalités spécifiques à la plateforme :
   ```go
   envInfo := app.Env.Info()
   if envInfo.OS == "darwin" {
       // Activer les fonctionnalités spécifiques à macOS
   }
   ```

## Exemple complet

Voici un exemple complet illustrant la gestion de l'environnement :

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
    "runtime"
    "github.com/wailsapp/wails/v3/pkg/application"
    "github.com/wailsapp/wails/v3/pkg/events"
)

func main() {
    app := application.New(application.Options{
        Name: "Démonstration Environnement",
    })

    // Configurer l'application en fonction de l'environnement
    setupForEnvironment(app)

    // Surveiller les changements de thème
    monitorThemeChanges(app)

    // Créer un menu avec les fonctionnalités de l'environnement
    setupEnvironmentMenu(app)

    // Créer la fenêtre principale
    window := app.Window.New()
    window.SetTitle("Démonstration Environnement")

    err := app.Run()
    if err != nil {
        panic(err)
    }
}

func setupForEnvironment(app *application.App) {
    envInfo := app.Env.Info()
    
    app.Logger.Info("Environnement de l'application",
        "os", envInfo.OS,
        "arch", envInfo.Arch,
        "debug", envInfo.Debug,
        "darkMode", app.Env.IsDarkMode(),
    )

    // Configurer pour la plateforme
    switch envInfo.OS {
    case "darwin":
        app.Logger.Info("Configuration pour macOS")
        // Configuration spécifique à macOS
    case "windows":
        app.Logger.Info("Configuration pour Windows")
        // Configuration spécifique à Windows
    case "linux":
        app.Logger.Info("Configuration pour Linux")
        // Configuration spécifique à Linux
    }

    // Appliquer le thème initial
    if app.Env.IsDarkMode() {
        applyDarkTheme(app)
    } else {
        applyLightTheme(app)
    }
}

func monitorThemeChanges(app *application.App) {
    app.Event.OnApplicationEvent(events.Common.ThemeChanged, func(event *application.ApplicationEvent) {
        if app.Env.IsDarkMode() {
            app.Logger.Info("Le système est passé en mode sombre")
            applyDarkTheme(app)
        } else {
            app.Logger.Info("Le système est passé en mode clair")
            applyLightTheme(app)
        }
    })
}

func setupEnvironmentMenu(app *application.App) {
    menu := app.Menu.New()
    
    // Ajouter le menu de l'application spécifique à la plateforme
    if runtime.GOOS == "darwin" {
        menu.AddRole(application.AppMenu)
    }
    
    // Menu Outils
    toolsMenu := menu.AddSubmenu("Outils")
    
    toolsMenu.Add("Afficher les informations de l'environnement").OnClick(func(ctx *application.Context) {
        showEnvironmentInfo(app)
    })
    
    toolsMenu.Add("Ouvrir le dossier Téléchargements").OnClick(func(ctx *application.Context) {
        openDownloadsFolder(app)
    })
    
    toolsMenu.Add("Basculer le thème").OnClick(func(ctx *application.Context) {
        // Cela serait généralement géré par le système
        // mais est montré ici à titre de démonstration
        toggleTheme(app)
    })
    
    app.Menu.Set(menu)
}

func showEnvironmentInfo(app *application.App) {
    envInfo := app.Env.Info()
    
    message := fmt.Sprintf(`Informations sur l'environnement :

Système d'exploitation : %s
Architecture : %s
Mode débogage : %t
Mode sombre : %t

Détails de la plateforme :`,
        envInfo.OS,
        envInfo.Arch,
        envInfo.Debug,
        app.Env.IsDarkMode())
    
    for key, value := range envInfo.PlatformInfo {
        message += fmt.Sprintf("\n%s : %v", key, value)
    }
    
    if envInfo.OSInfo != nil {
        message += fmt.Sprintf("\n\nInformations sur le système d'exploitation :\nNom : %s\nVersion : %s",
            envInfo.OSInfo.Name,
            envInfo.OSInfo.Version)
    }
    
    dialog := app.Dialog.Info()
    dialog.SetTitle("Informations sur l'environnement")
    dialog.SetMessage(message)
    dialog.Show()
}

func openDownloadsFolder(app *application.App) {
    homeDir, err := os.UserHomeDir()
    if err != nil {
        app.Logger.Error("Impossible d'obtenir le répertoire personnel", "error", err)
        return
    }
    
    downloadsDir := filepath.Join(homeDir, "Téléchargements")
    err = app.Env.OpenFileManager(downloadsDir, false)
    if err != nil {
        app.Logger.Error("Impossible d'ouvrir le dossier Téléchargements", "error", err)
        
        app.Dialog.Error().
            SetTitle("Erreur du gestionnaire de fichiers").
            SetMessage("Impossible d'ouvrir le dossier Téléchargements").
            Show()
    }
}

func applyDarkTheme(app *application.App) {
    app.Logger.Info("Application du thème sombre")
    // Émettre le changement de thème vers le frontend
    app.Event.Emit("theme:apply", "dark")
}

func applyLightTheme(app *application.App) {
    app.Logger.Info("Application du thème clair")
    // Émettre le changement de thème vers le frontend
    app.Event.Emit("theme:apply", "light")
}

func toggleTheme(app *application.App) {
    // Ceci est uniquement à titre de démonstration
    // Les vrais changements de thème doivent provenir du système
    currentlyDark := app.Env.IsDarkMode()
    if currentlyDark {
        applyLightTheme(app)
    } else {
        applyDarkTheme(app)
    }
}:::tip[Conseil]
Utilisez les informations de l’environnement pour offrir des expériences utilisateur adaptées à chaque plateforme. Par exemple, utilisez les raccourcis clavier avec la touche Command sur macOS et avec la touche Control sur Windows/Linux.
:::

:::danger[Attention]
Les informations de l’environnement sont généralement stables pendant l’exécution de l’application, mais les préférences de thème peuvent changer. Écoutez toujours les événements de changement de thème pour maintenir votre interface utilisateur synchronisée.
:::
